## Install

### Install from `npm`:

```bash
$ npm install artplayer-plugin-ads
```

### Install from `yarn`:

```bash
$ yarn add artplayer-plugin-ads
```

### Install from `script`:

```html
<!-- local -->
<script src="path/to/artplayer-plugin-ads.js"></script>

<!-- jsdelivr -->
<script src="https://cdn.jsdelivr.net/npm/artplayer-plugin-ads/dist/artplayer-plugin-ads.js"></script>

<!-- unpkg -->
<script src="https://unpkg.com/artplayer-plugin-ads/dist/artplayer-plugin-ads.js"></script>
```

Then you can access the plugin function through `window.artplayerPluginAds`

## Use

```js
var art = new Artplayer({
    container: '.artplayer-app',
    url: '/assets/sample/video.mp4',
    autoSize: true,
    fullscreen: true,
    fullscreenWeb: true,
    plugins: [
        artplayerPluginAds({
            // html ad, ignore this value if it is a video ad
            html: '<img src="/assets/sample/poster.jpg">',

            // The address of the video ad
            video: '/assets/sample/test1.mp4',

            // Ad redirect URL, if empty, no redirect
            url: 'http://artplayer.org',

            // The length of time that must be watched, the period cannot be skipped, the unit is seconds
            playDuration: 5,

            // The total duration of the ad, in seconds
            totalDuration: 10,

            // Multilingual support
            i18n: {
                close: 'Close Ads',
                countdown: '%s seconds',
                detail: 'Check Detail',
                canBeClosed: 'Ads can be closed after %s seconds',
            },
        }),
    ],
});

art.on('artplayerPluginAds:click', (ads) => {
    console.info('ad is clicked', ads);
});

art.on('artplayerPluginAds:skip', (ads) => {
    console.info('Ads are skipped', ads);
});
```
